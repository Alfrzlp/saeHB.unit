model {
					for (j in 1:J) {
							y[j] ~ dnorm(Mu[j],tau.e)
							Mu[j] <- b0+b1*x[j]+u[d[j]]
							
					}		
					
				for(i in 1:m){
					u[i] ~ dnorm(0,tau.u)
					mu[i] <- b0+b1*X[i]+u[i]
					}
				  
				  b0 ~ dnorm(mu.b0,t.b0)
					b1 ~ dnorm(mu.b1,t.b1)
					
					tau.e ~ dgamma(tau.ea,tau.eb)
					tau.u ~ dgamma(tau.ua,tau.ub)
					a.var <- 1 / tau.u
